SELECT
  COUNT(*) FILTER (WHERE content_enriched IS NOT NULL AND content_enriched <> '') AS enriched,
  COUNT(*) AS total,
  ROUND(
    100.0 * COUNT(*) FILTER (WHERE content_enriched IS NOT NULL AND content_enriched <> '') / COUNT(*),
    2
  ) AS percent_complete
FROM raw_content
WHERE cluster_id = 'hips'
  AND collection_id = 'hazards'
  AND version = '2.0.0';
